//TORRES DE HANOI

hanoi(n,origen,auxiliar,destino)
begin
	if(n = 1)then
	begin
		print('mover disco del '+ origen +'al '+destino)
	end
	else
	begin
		call hanoi(n-1,origen,destino,auxiliar)
		print('mover disco del '+ origen +'al '+destino)
		call hanoi(n-1,auxiliar,origen,destino)
	end
end
hanoi(5, 'o', 'a', 'd')




// BUSQUEDA BINARIA RECURSIVA

bbr(a[],x,liminf,limsup)
begin
	pos←1
	if(limsup > liminf)then
	begin
		mitad←(limsup+liminf)/2
		if(a[mitad] = x)then
		begin
			pos←mitad
		end
		else
		begin
			if(a[mitad] > x)then
			begin
				pos←bbr(a,x,liminf,mitad-1)
			end
			else
			begin
				pos←bbr(a,x,mitad+1,limsup-1)
			end
		end
	end
	↑pos
end



//FACTORIAL

factorial(n)
begin
	f←0
	if((n=0) or (n=1))then
	begin
		f←1
	end
	else
		begin
			f←n*(call factorial(n-1))
		end
	↑f
end
factorial(20)





//MERGESORT

merge(A,p,q,r)
begin
	n1←q-p+1
	n2←r-q
	L[n1+1]
	R[n2+1]
	for i ← 1 to n1 UPTO(1) do
	begin
		L[i]←A[p+i-1]
	end
	for j ← 1 to n1 UPTO(1) do
	begin
		R[j]←A[q+j]
	end
	L[n1+1]←inf
	R[n2+1]←inf
	i←1
	j←1
	for k ← p to r UPTO(1) do
	begin
		if L[i] ≤ R[j] then
		begin
			a[k]←L[i]
			i←i+1
		end
		else
		begin
			A[k]←L[i]
			j←j+1
		end
	end
end

merge_sort(A, p, r)
begin
	if(p < r) then 
	begin
		q←⌊(p+r)/2⌋
		merge_sort(A,p,q)
		merge_sort(A,q+1,r)
		merge(A,p,q,r)
	end
end
